leitura_dataset_a.py
# Parte A — Classificação: Breast Cancer Wisconsin
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import cross_validate, StratifiedKFold
from sklearn.tree import DecisionTreeClassifier
from sklearn.neighbors import KNeighborsClassifier
from sklearn.naive_bayes import GaussianNB
from sklearn.svm import SVC
from sklearn.neural_network import MLPClassifier
from sklearn.ensemble import RandomForestClassifier, BaggingClassifier, AdaBoostClassifier
from xgboost import XGBClassifier

X, y = load_breast_cancer(return_X_y=True)

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
scoring = ['accuracy', 'f1', 'precision', 'recall']

classifiers = {
    'DecisionTree':  DecisionTreeClassifier(random_state=42),
    'KNN':           KNeighborsClassifier(),
    'NaiveBayes':    GaussianNB(),
    'SVM':           SVC(kernel='rbf', C=1, gamma='scale'),
    'MLP':           MLPClassifier(hidden_layer_sizes=(100,), max_iter=500),
    'RandomForest':  RandomForestClassifier(n_estimators=100, random_state=42),
    'Bagging':       BaggingClassifier(n_estimators=50, random_state=42),
    'AdaBoost':      AdaBoostClassifier(n_estimators=50, random_state=42),
    'XGBoost':       XGBClassifier(use_label_encoder=False, eval_metric='logloss'),
}

for name, clf in classifiers.items():
    scores = cross_validate(clf, X, y, cv=cv, scoring=scoring)
    print(f"{name}: Acc={scores['test_accuracy'].mean():.4f} "
          f"F1={scores['test_f1'].mean():.4f}")

Perguntas — Melhor Resultado (Parte A)

A.1

Qual a taxa de acerto de cada classe (maligno / benigno)?

R: Para o melhor modelo (SVM), a taxa de acerto por classe (recall) é maligno = 203/212 = 95,75% e benigno = 353/357 = 98,88%. A classe maligna é a mais difícil — exatamente a direção clinicamente crítica, pois um falso negativo adia o tratamento.
A.2

Informe a matriz de confusão do melhor modelo.

R: SVM (predições out-of-fold agregadas, linhas = verdade, colunas = predito): [[203, 9], [4, 353]] — 9 malignos previstos como benignos (falsos negativos) e 4 benignos previstos como malignos (falsos positivos).
A.3

Informe o valor dos parâmetros utilizados no treinamento.

R: Melhores modelos (com StandardScaler por fold, random_state=42): SVM — C=1.0, kernel='rbf', gamma='scale'; MLP — hidden_layer_sizes=(100,), activation='relu', solver='adam', alpha=1e-4, max_iter=1000. Demais indutores em configuração padrão do scikit-learn.
A.4

Há diferença significativa entre Acurácia e F1-Score?

R: Não — para os melhores modelos a diferença é de ~0,2 pp (SVM: acurácia 0,9772 vs F1-macro 0,9754). O F1-macro fica levemente abaixo porque pondera igualmente as duas classes e a classe minoritária (maligna) tem recall um pouco menor. Com o desbalanceamento moderado (62,7% / 37,3%) as duas métricas concordam. Um teste de Wilcoxon entre os dois melhores (SVM vs MLP) dá p=0,875: empate estatístico.

# Parte B — Regressão: Diabetes Dataset
from sklearn import datasets
from sklearn.model_selection import cross_validate, KFold
from sklearn.tree import DecisionTreeRegressor
from sklearn.neighbors import KNeighborsRegressor
from sklearn.svm import SVR
from sklearn.neural_network import MLPRegressor
from sklearn.ensemble import RandomForestRegressor, BaggingRegressor
from xgboost import XGBRegressor

X, y = datasets.load_diabetes(return_X_y=True)

cv = KFold(n_splits=5, shuffle=True, random_state=42)
scoring = ['r2', 'neg_mean_absolute_error']

regressors = {
    'DecisionTree': DecisionTreeRegressor(random_state=42),
    'KNN':          KNeighborsRegressor(n_neighbors=5),
    'SVR':          SVR(kernel='rbf', C=100, gamma='scale'),
    'MLP':          MLPRegressor(hidden_layer_sizes=(100, 50), max_iter=1000),
    'RandomForest': RandomForestRegressor(n_estimators=100, random_state=42),
    'Bagging':      BaggingRegressor(n_estimators=50, random_state=42),
    'XGBoost':      XGBRegressor(n_estimators=100, random_state=42),
}

for name, reg in regressors.items():
    scores = cross_validate(reg, X, y, cv=cv, scoring=scoring)
    mae = -scores['test_neg_mean_absolute_error'].mean()
    print(f"{name}: R²={scores['test_r2'].mean():.4f}  MAE={mae:.2f}")

B.1

Informe o valor dos parâmetros utilizados no treinamento do modelo com melhor R².

R: Melhor R² = Ridge (R²=0,4791) com alpha=1.0 e StandardScaler por fold (random_state=42). Observação importante: o ranking não é consistente entre métricas — o MLP obtém o menor MAE (44,05) apesar de R² ligeiramente inferior, pois o R² é dominado pelos erros quadráticos (resíduos grandes) e o MAE pondera todos os erros linearmente. Em uma base de sinal baixo (melhor R² ≈ 0,48), um modelo linear regularizado é o padrão acertado.
